31  public static string Encrypt( string plainText, string passPhrase ) {
32  // Salt and IV is randomly generated each time, but is preprended to encrypted cipher text
33  // so that the same Salt and IV values can be used when decrypting.
34  byte[] saltStringBytes = SimpleStringCipher.Generate256BitsOfRandomEntropy();
35  byte[] ivStringBytes = SimpleStringCipher.Generate256BitsOfRandomEntropy();
36  byte[] plainTextBytes = System.Text.Encoding.UTF8.GetBytes( plainText );
37 
38  using( var password = new Rfc2898DeriveBytes( passPhrase, saltStringBytes, SimpleStringCipher.DerivationIterations ) ) {
39  byte[] keyBytes = password.GetBytes( Keysize / 8 );
40 
41  using( var symmetricKey = new RijndaelManaged() ) {
42  symmetricKey.BlockSize = 256;
43  symmetricKey.Mode = CipherMode.CBC;
44  symmetricKey.Padding = PaddingMode.PKCS7;
45 
46  using( var encryptor = symmetricKey.CreateEncryptor( keyBytes, ivStringBytes ) ) {

49  cryptoStream.Write( plainTextBytes, 0, plainTextBytes.Length );
50  cryptoStream.FlushFinalBlock();
51

73  public static string Decrypt( string cipherText, string passPhrase ) {
74  // Get the complete stream of bytes that represent:
75  // [32 bytes of Salt] + [32 bytes of IV] + [n bytes of CipherText]
76  byte[] cipherTextBytesWithSaltAndIv = Convert.FromBase64String( cipherText );
77  // Get the saltbytes by extracting the first 32 bytes from the supplied cipherText bytes.
78  byte[] saltStringBytes = cipherTextBytesWithSaltAndIv.Take( Keysize / 8 ).ToArray();
79  // Get the IV bytes by extracting the next 32 bytes from the supplied cipherText bytes.
80  byte[] ivStringBytes = cipherTextBytesWithSaltAndIv.Skip( Keysize / 8 ).Take( Keysize / 8 ).ToArray();
81  // Get the actual cipher text bytes by removing the first 64 bytes from the cipherText string.
82  byte[] cipherTextBytes = cipherTextBytesWithSaltAndIv.Skip( ( Keysize / 8 ) * 2 ).Take( cipherTextBytesWithSaltAndIv.Length - ( ( Keysize / 8 ) * 2 ) ).ToArray();
83 
84  using( var password = new Rfc2898DeriveBytes( passPhrase, saltStringBytes, SimpleStringCipher.DerivationIterations ) ) {
85  byte[] keyBytes = password.GetBytes( Keysize / 8 );
86 
87  using( var symmetricKey = new RijndaelManaged() ) {
88  symmetricKey.BlockSize = 256;
89  symmetricKey.Mode = CipherMode.CBC;
90  symmetricKey.Padding = PaddingMode.PKCS7;
91 
92  using( var decryptor = symmetricKey.CreateDecryptor( keyBytes, ivStringBytes ) ) {
93  using( var memoryStream = new MemoryStream( cipherTextBytes ) ) {
94  using( var cryptoStream = new CryptoStream( memoryStream, decryptor, CryptoStreamMode.Read ) ) {
95  var plainTextBytes = new byte[cipherTextBytes.Length];
96  int decryptedByteCount = cryptoStream.Read( plainTextBytes, 0, plainTextBytes.Length );
97 
98  memoryStream.Close();
99  cryptoStream.Close();
100  return System.Text.Encoding.UTF8.GetString( plainTextBytes, 0, decryptedByteCount );
101  }
