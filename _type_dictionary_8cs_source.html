78  public TypeData this[Type key]

83

85 
86  Pair[] arr = Volatile.Read(ref buckets[idx]);

89 
90  for (int i = 0; i < arr.Length; ++i)
91  {

94  }
95 
96  throw new KeyNotFoundException(String.Format("Type not found {0}", key));
97  }

99  set
100  {
101  lock (m_writeLock)
102  {
103  Debug.Assert(ContainsKey(key) == false);
104 
105  if (m_numItems >= m_buckets.Length * LoadLimit)
106  {
107  var newBuckets = new Pair[m_buckets.Length * 2][];
108 
109  foreach (var list in m_buckets.Where(l => l != null))
110  {
111  foreach (var pair in list.Where(p => p.Key != null))
112  Add(newBuckets, pair.Key, pair.Value);
113  }
114 
115  Volatile.Write(ref m_buckets, newBuckets);
116  }
117 
118  Add(m_buckets, key, value);
119 
120  m_numItems++;
121  }
122  }

125  static void Add(Pair[][] buckets, Type key, TypeData value)
126  {
127  int idx = Hash(key, buckets.Length);
128 
129  Pair[] arr = buckets[idx];
130  if (arr == null)
131  buckets[idx] = arr = new Pair[InitialListSize];
132 
133  for (int i = 0; i < arr.Length; ++i)
134  {
135  if (arr[i].Key == null)
136  {
137  arr[i] = new Pair(key, value);
138  return;
139  }
140  }
141 
142  var newArr = new Pair[arr.Length * 2];
143  Array.Copy(arr, newArr, arr.Length);
144  newArr[arr.Length] = new Pair(key, value);
145  buckets[idx] = newArr;
146  }
