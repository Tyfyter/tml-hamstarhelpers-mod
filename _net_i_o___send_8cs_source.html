116  private static bool Send( NetIOPayload data, int toWho, int ignoreWho ) {
117  var netIO = ModContent.GetInstance<NetIO>();
118  ModPacket packet = ModHelpersMod.Instance.GetPacket();
119 
120  try {
121  Type dataType = data.GetType();
122  if( !netIO.PayloadTypeToCode.TryGetValue(dataType, out int code) ) {
123  return false;
124  }
125 
126  Serializer ser = netIO.PayloadCodeToSerializer[ code ];
127 
128  packet.Write( (int)code );
129 
130  MethodInfo method = ser.GetType().GetMethod( "SerializeDirect", ReflectionHelpers.MostAccess );
131  method = method.MakeGenericMethod( new Type[] { dataType } );
132  method.Invoke( ser, new object[] { packet.BaseStream, data } );
133 
134  packet.Send( toWho, ignoreWho );
135 
136  if( ModHelpersConfig.Instance.DebugModeNetInfo ) {
137  LogHelpers.Log( ">" + dataType.Name + " "+toWho+" "+ignoreWho );
138  }
139 
140  return true;
141  } catch( Exception e ) {
142  LogHelpers.WarnOnce( e.ToString() );
143  }
144
