53  public static int RemoveInventoryItemQuantity( Player player, int itemType, int quantity ) {
54  int removed = 0;
55 
56  for( int i = 0; i < player.inventory.Length; i++ ) {
57  Item item = player.inventory[i];
58  if( item == null || item.IsAir || item.type != itemType ) { continue; }
59 
60  int stack = item.stack;
61 
62  if( stack > quantity ) {
63  item.stack -= quantity;
64  removed += quantity;
65  break;
66  } else {
67  quantity -= stack;
68  removed += stack;
69  player.inventory[i] = new Item();
70 
71  if( quantity == 0 ) { break; }
72  }
73  }
74 
75  return removed;
76  }

214  for( int i = 0; i < player.inventory.Length; i++ ) {
215  if( player.inventory[i] == null || player.inventory[i].IsAir ) {
216  player.selectedItem = i;
217  isUnhanded = true;
218  break;
219  }
220  }
221  }
222  // Otherwise select a non-usable item
223  if( !isUnhanded ) {
224  for( int i = 0; i < player.inventory.Length; i++ ) {
225  Item item = player.inventory[i];
226  if( item != null && item.holdStyle == 0 && item.createTile == -1 && !item.potion && item.useStyle == 0 ) {
227  player.selectedItem = i;
228  isUnhanded = true;
229  break;
230  }
231  }
232  }
233  // Otherwise select a non-held item
234  if( !isUnhanded ) {
235  for( int i = 12; i < player.inventory.Length; i++ ) {
236  Item item = player.inventory[i];
237  if( item != null && item.holdStyle == 0 ) {
238  player.selectedItem = i;
239  isUnhanded = true;

241  }
242  }
243  }
244 
245  player.noItems = true;
246 
247  return isUnhanded;
