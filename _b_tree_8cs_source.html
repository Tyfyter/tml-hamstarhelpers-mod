116  private void DeleteKeyFromSubtree( BTreeNode<TK, TP> parentNode, TK keyToDelete, int subtreeIndexInNode ) {
117  BTreeNode<TK, TP> childNode = parentNode.Children[subtreeIndexInNode];
118 
119  // node has reached min # of entries, and removing any from it will break the btree property,
120  // so this block makes sure that the "child" has at least "degree" # of nodes by moving an
121  // entry from a sibling node or merging nodes
122  if( childNode.HasReachedMinEntries ) {
123  int leftIndex = subtreeIndexInNode - 1;
124  BTreeNode<TK, TP> leftSibling = subtreeIndexInNode > 0 ? parentNode.Children[leftIndex] : null;
125 
126  int rightIndex = subtreeIndexInNode + 1;
127  BTreeNode<TK, TP> rightSibling = subtreeIndexInNode < parentNode.Children.Count - 1
128  ? parentNode.Children[rightIndex]
129  : null;
130 
131  if( leftSibling != null && leftSibling.Entries.Count > this.Degree - 1 ) {
132  // left sibling has a node to spare, so this moves one node from left sibling
133  // into parent's node and one node from parent into this current node ("child")
134  childNode.Entries.Insert( 0, parentNode.Entries[subtreeIndexInNode] );
135  parentNode.Entries[subtreeIndexInNode] = leftSibling.Entries.Last();
136  leftSibling.Entries.RemoveAt( leftSibling.Entries.Count - 1 );
137 
138  if( !leftSibling.IsLeaf ) {
139  childNode.Children.Insert( 0, leftSibling.Children.Last() );
140  leftSibling.Children.RemoveAt( leftSibling.Children.Count - 1 );
141  }
142  } else if( rightSibling != null && rightSibling.Entries.Count > this.Degree - 1 ) {
143  // right sibling has a node to spare, so this moves one node from right sibling
144  // into parent's node and one node from parent into this current node ("child")
145  childNode.Entries.Add( parentNode.Entries[subtreeIndexInNode] );
146  parentNode.Entries[subtreeIndexInNode] = rightSibling.Entries.First();
147  rightSibling.Entries.RemoveAt( 0 );
148 
149  if( !rightSibling.IsLeaf ) {
150  childNode.Children.Add( rightSibling.Children.First() );
151  rightSibling.Children.RemoveAt( 0 );
152  }
153  } else {
154  // this block merges either left or right sibling into the current node "child"
155  if( leftSibling != null ) {
156  childNode.Entries.Insert( 0, parentNode.Entries[subtreeIndexInNode] );
157  var oldEntries = childNode.Entries;
158  childNode.Entries = leftSibling.Entries;
159  childNode.Entries.AddRange( oldEntries );
160  if( !leftSibling.IsLeaf ) {
161  var oldChildren = childNode.Children;
162  childNode.Children = leftSibling.Children;
163  childNode.Children.AddRange( oldChildren );
164  }
165 
166  parentNode.Children.RemoveAt( leftIndex );
167  parentNode.Entries.RemoveAt( subtreeIndexInNode );
168  } else {
169  Debug.Assert( rightSibling != null, "Node should have at least one sibling" );
170  childNode.Entries.Add( parentNode.Entries[subtreeIndexInNode] );
171  childNode.Entries.AddRange( rightSibling.Entries );
172  if( !rightSibling.IsLeaf ) {
173  childNode.Children.AddRange( rightSibling.Children );
174  }
175 
176  parentNode.Children.RemoveAt( rightIndex );
177  parentNode.Entries.RemoveAt( subtreeIndexInNode );
178  }
179  }
180  }
181 
182  // at this point, we know that "child" has at least "degree" nodes, so we can
183  // move on - this guarantees that if any node needs to be removed from it to
184  // guarantee BTree's property, we will be fine with that
185  this.DeleteInternal( childNode, keyToDelete );
186  }

195  private void DeleteKeyFromNode( BTreeNode<TK, TP> node, TK keyToDelete, int keyIndexInNode ) {
196  // if leaf, just remove it from the list of entries (we're guaranteed to have
197  // at least "degree" # of entries, to BTree property is maintained
198  if( node.IsLeaf ) {
199  node.Entries.RemoveAt( keyIndexInNode );
200  return;
201  }
202 
203  BTreeNode<TK, TP> predecessorChild = node.Children[keyIndexInNode];
204  if( predecessorChild.Entries.Count >= this.Degree ) {
205  BTreeEntry<TK, TP> predecessor = this.DeletePredecessor( predecessorChild );
206  node.Entries[keyIndexInNode] = predecessor;
207  } else {
208  BTreeNode<TK, TP> successorChild = node.Children[keyIndexInNode + 1];
209  if( successorChild.Entries.Count >= this.Degree ) {
210  BTreeEntry<TK, TP> successor = this.DeleteSuccessor( predecessorChild );
211  node.Entries[keyIndexInNode] = successor;
212  } else {
213  predecessorChild.Entries.Add( node.Entries[keyIndexInNode] );
214  predecessorChild.Entries.AddRange( successorChild.Entries );
215  predecessorChild.Children.AddRange( successorChild.Children );
216 
217  node.Entries.RemoveAt( keyIndexInNode );
218  node.Children.RemoveAt( keyIndexInNode + 1 );
219 
220  this.DeleteInternal( predecessorChild, keyToDelete );
221  }
222  }

296
