179  public static IDictionary<int, int> FindChanges( Item[] prevItems, Item[] currItems, bool skipCoins=true ) {
180  if( prevItems.Length != currItems.Length ) {
181  throw new ModHelpersException( "Mismatched item array sizes." );
182  }
183 
184  IDictionary<int, int> changes = new Dictionary<int, int>();
185  int len = currItems.Length;
186  Item prevItem, currItem;
187 
188  for( int i = 0; i < len; i++ ) {
189  prevItem = prevItems[i];
190  currItem = currItems[i];
191 
192  if( prevItem == null ) {
193  if( currItem != null ) {
194  changes[i] = 1;
195  }
196  continue;
197  } else if( currItem == null ) {
198  if( prevItem != null ) {
199  changes[i] = -1;
200  }
201  continue;
202  }
203 
204  if( (prevItem.type != currItem.type || prevItem.stack != currItem.stack) && prevItem.IsAir != currItem.IsAir ) {
205  if( prevItem.IsAir ) {
206  changes[i] = 1;
207  } else if( currItem.IsAir ) {
208  changes[i] = -1;
209  } else {
210  changes[i] = prevItem.stack < currItem.stack ? 1 : -1;
211  }
212  }
213  }
214 
215  if( skipCoins ) {
216  foreach( (int itemIdx, int changeDir) in changes.ToArray() ) {
217  currItem = currItems[itemIdx];
218  prevItem = prevItems[itemIdx];
219 
220  if( currItem == null || currItem.IsAir ) {
221  if( prevItem.type >= ItemID.CopperCoin && prevItem.type <= ItemID.PlatinumCoin ) {
222  changes.Remove( itemIdx );
223  continue;
224  }
225  }
226 
227  if( currItems[itemIdx].type >= ItemID.CopperCoin || currItems[itemIdx].type <= ItemID.PlatinumCoin ) {
228  changes.Remove( itemIdx );
229  continue;
230  }
231  }
232  }
233 
234  return changes;
235  }
