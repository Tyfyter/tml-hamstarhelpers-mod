38  public bool Apply( UIElement element ) {
39  foreach( CustomAttributeData attr in element.GetType().CustomAttributes ) {
40  if( !attr.AttributeType.IsSubclassOf(typeof(ThemedAttrbute)) ) {
41  continue;
42  }
43 
44  if( attr.AttributeType == typeof( PanelThemeAttribute ) ) {
45  if( element is UIPanel ) {
46  this.ApplyPanel( (UIPanel)element );
47  return true;
48  }
49  } else if( attr.AttributeType == typeof( TextThemeAttribute ) ) {
50  if( element is UIText ) {
51  this.ApplyText( (UIText)element );
52  return true;
53  }
54  } else if( attr.AttributeType == typeof(ListContainerThemeAttribute) ) {
55  if( element is UIPanel ) {
56  this.ApplyListContainer( (UIPanel)element );
57  return true;
58  }
59  } else if( attr.AttributeType == typeof(ListItemThemeAttribute) ) {
60  if( element is UIPanel ) {
61  this.ApplyListItem( (UIPanel)element );
62  return true;
63  }
64  } else if( attr.AttributeType == typeof(ButtonThemeAttribute) ) {
65  if( element is UITextPanel<string> ) {
66  this.ApplyButton( (UITextPanel<string>)element );
67  return true;
68  }
69  } else if( attr.AttributeType == typeof(InputThemeAttribute) ) {
70  if( element is UITextInputPanel ) {
71  this.ApplyInput( (UITextInputPanel)element );
72  return true;
73  }
74  }
75 
76  throw new ModHelpersException( "Invalid theme Attribute "+attr.AttributeType.Name+" for "+element.GetType().Name );
77  }
78 
79  if( element is IThemeable ) {
80  ((IThemeable)element).SetTheme( this );
81  return true;
82  }
83 
84  return false;
85  }
