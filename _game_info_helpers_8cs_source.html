186  public static IDictionary<string, string> GetPlayerEquipment( Player player ) {

197  if( i == 0 ) {
198  key = "Head";
199  } else if( i == 1 ) {
200  key = "Body";
201  } else if( i == 2 ) {
202  key = "Legs";
203  } else if( PlayerItemHelpers.IsAccessorySlot( player, i ) ) {
204  key = "Accessory "+acc;
205  acc++;
206  } else if( PlayerItemHelpers.IsVanitySlot( player, i ) ) {
207  key = "Vanity "+van;

209  } else {
210  key = "? "+unk;
211  unk++;
212  }
213
