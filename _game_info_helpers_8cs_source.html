209  public static IList<string> GetErrorLog( int maxLines ) {
210  if( maxLines > 150 ) { maxLines = 150; }
211 
212  IList<string> lines = new List<string>();
213  char sep = Path.DirectorySeparatorChar;
214  string path = Main.SavePath + sep + "Logs" + sep + "Logs.txt";
215 
216  if( !File.Exists( path ) ) {
217  return new List<string> { "No error logs available." };
218  }
219 
220  using( var reader = new StreamReader( path ) ) {
221  int size = 1024;
222  bool eof = false;
223 
224  do {
225  lines = new List<string>( maxLines + 25 );
226 
227  if( reader.BaseStream.Length > size ) {
228  reader.BaseStream.Seek( -size, SeekOrigin.End );
229  size += 1024;
230  } else {
231  eof = true;
232  reader.BaseStream.Seek( 0, SeekOrigin.Begin );
233  }
234 
235  string line;
236  while( ( line = reader.ReadLine() ) != null ) {
237  lines.Add( line );
238  }
239  } while( lines.Count < maxLines && !eof );
240  }
241 
242  IList<string> revLines = lines.Reverse().Take( 25 ).ToList();
243  if( lines.Count > maxLines ) { revLines.Add( "..." ); }
244 
245  return new List<string>( revLines.Reverse() );
246  }
