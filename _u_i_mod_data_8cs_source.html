137  return 1;

139 
140  // Always sort own mod to top; this mod's configs should be available first
141  if( other.Mod.Name == ModHelpersMod.Instance.Name ) {
142  return 1;
143  } else if( this.Mod.Name == ModHelpersMod.Instance.Name ) {
144  return -1;

146 
147  try {

149  if( ModFeaturesHelpers.HasGithub( this.Mod ) && !ModFeaturesHelpers.HasGithub( other.Mod ) ) {
150  return -1;
151  } else if( !ModFeaturesHelpers.HasGithub( this.Mod ) && ModFeaturesHelpers.HasGithub( other.Mod ) ) {
152  return 1;
153  }
154 
155  // Prioritize config'd mods

157  return -1;

159  return 1;
