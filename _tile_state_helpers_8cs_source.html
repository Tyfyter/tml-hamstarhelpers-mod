152  public static bool SmartSlope( int tileX, int tileY, bool isSynced ) {
153  Tile mid = Framing.GetTileSafely( tileX, tileY );
154  Tile up = Framing.GetTileSafely( tileX, tileY - 1 );
155  Tile down = Framing.GetTileSafely( tileX, tileY + 1 );
156  Tile left = Framing.GetTileSafely( tileX - 1, tileY );
157  Tile right = Framing.GetTileSafely( tileX + 1, tileY );
158 
159  bool upSolid = up.active() && up.slope() == 0 && Main.tileSolid[up.type] && !Main.tileSolidTop[up.type];
160  bool downSolid = down.active() && down.slope() == 0 && Main.tileSolid[down.type] && !Main.tileSolidTop[down.type];
161  bool leftSolid = left.active() && left.slope() == 0 && Main.tileSolid[left.type] && !Main.tileSolidTop[left.type];
162  bool rightSolid = right.active() && right.slope() == 0 && Main.tileSolid[right.type] && !Main.tileSolidTop[right.type];
163 
164  upSolid = upSolid
165  && (up.slope() == (byte)TileSlopeType.TopLeftSlope || up.slope() == (byte)TileSlopeType.TopRightSlope);
166  downSolid = downSolid
167  && (down.slope() == (byte)TileSlopeType.BottomLeftSlope || down.slope() == (byte)TileSlopeType.BottomRightSlope );
168  leftSolid = leftSolid
169  && (left.slope() == (byte)TileSlopeType.TopRightSlope || left.slope() == (byte)TileSlopeType.BottomRightSlope );
170  rightSolid = rightSolid
171  && ( right.slope() == (byte)TileSlopeType.TopLeftSlope || right.slope() == (byte)TileSlopeType.BottomLeftSlope );
172  int changed = 0;
173 
174  // Up
175  if( !upSolid && downSolid && leftSolid && !rightSolid ) {
176  mid.slope( (byte)TileSlopeType.TopLeftSlope );
177  changed = 1;
178  }
179  if( !upSolid && downSolid && !leftSolid && rightSolid ) {
180  mid.slope( (byte)TileSlopeType.TopRightSlope );
181  changed = 1;
182  }
183  if( !upSolid && downSolid && !leftSolid && !rightSolid ) {
184  mid.halfBrick( true );
185  changed = 2;
186  }
187 
188  // Down
189  if( upSolid && !downSolid && leftSolid && !rightSolid ) {
190  mid.slope( (byte)TileSlopeType.BottomLeftSlope );
191  changed = 1;
192  }
193  if( upSolid && !downSolid && !leftSolid && rightSolid ) {
194  mid.slope( (byte)TileSlopeType.BottomRightSlope );
195  changed = 1;
196  }
197 
198  if( isSynced && Main.netMode == NetmodeID.MultiplayerClient ) {
199  if( changed != 0 ) {
200  NetMessage.SendData( MessageID.TileChange, -1, -1, null, (int)TileChangeNetMessageType.SlopeTile, (float)tileX, (float)tileY, (float)mid.slope(), 0, 0, 0 );
201  }
202  }
203 
204  return false;
205  }
